>[!WARNING|style: flat|label:  简要说明 ]
>
>- 原始的配置数据通过`IConfigurationProvider`对象转换成<span style='color:Blue'>[ 中间 - 配置字典 ]</span> (`此接口中的方法大部分体现对字典的操作`)
>
>```csharp
>public interface IConfigurationProvider
>{
>
>       //[ 模板方法 ]加载转换配置数据
>       void Load();
>
>       void Set(string key, string value);
>       bool TryGet(string key, out string value);
>
>       IEnumerable<string> GetChildKeys(IEnumerable<string> earlierKeys, string parentPath);
>       //[ 监听 ]数据来源变更
>       IChangeToken GetReloadToken();
>    
>}
>
>
>```
>
>```csharp
>//默认实现
>public abstract class ConfigurationProvider : IConfigurationProvider
>{
>
>       //[ 中间字典 ]配置数据
>       protected IDictionary<string, string> Data { get; set; }
>       protected ConfigurationProvider()
>       {
>            this.Data = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
>       }
>
>       public virtual void Set(string key, string value) => this.Data[key] = value;
>       public virtual bool TryGet(string key, out string value) => this.Data.TryGetValue(key, out value);
> 
>       //[ 模板方法 ]加载转换配置数据
>       public virtual void Load()
>       { }
> 
>       #region [ 重载 ]数据来源变更
>
>       private ConfigurationReloadToken _reloadToken = new ConfigurationReloadToken();
>       public IChangeToken GetReloadToken() => this._reloadToken;
> 
>       protected void OnReload()
>       {
>            ConfigurationReloadToken previousToken = Interlocked.Exchange(ref _reloadToken, new ConfigurationReloadToken());
>            previousToken.OnReload();
>       }
> 
>       #endregion
> 
>       public virtual IEnumerable<string> GetChildKeys(IEnumerable<string> earlierKeys, string parentPath);    
> 
> }
> 
> 
> ```
