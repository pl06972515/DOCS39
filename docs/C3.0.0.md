#### **[ 脚本参数 ]**

!> <span style='color:red'>[ 建议：先通过普通任务测试，在集成`Jenkinsfile`]</span> [<span style='color:#008B00'>[👓 Q1 ]</span>](T1.0.0 ':target=_blank')

```csharp
pipeline {
    agent any 
    
    options {
        
        // 保留最近 30 次构建( 生产需要更多回滚记录 )
        buildDiscarder(logRotator(numToKeepStr: '30'))
        // 禁止并发构建 ( 但不中止旧构建 - 排队等待 ）
        disableConcurrentBuilds()
        // 构建超时保护（ 15 分钟 ）
        timeout(time: 15, unit: 'MINUTES')

    }
    
    # [ 全局: 环境变量 ]
    # - 用途：常用于配置、凭据、路径等
    # - 访问方式：${env.变量名} ( 在步骤 sh 中直接使用 ${变量名} )
    environment {
        EN1 = 'Q1'
        EN2 = 'Q2'
    }
    
    # [ 用户参数 ]
    # - 用途：在手动或定时触发流水线时 [ 允许用户输入或选择参数- 如分支、版本号、姓名等 ]
    # - 设置时机：在流水线启动时由用户填写或选择
    # - 访问方式：${params.参数名} ( 在步骤 sh 中无法直接使用 - 需通过环境变量 )
    parameters {
        string(name:"Name", defaultValue:"SYL", description: "姓名")
        string(name: "Age", defaultValue: "18", description: "年龄（请输入数字）")
    }
    
    stages {
    
        # [ 阶段定义 ]
        stage('第一阶段') {
            # [ 局部: 环境变量 ] 
            environment {
                EN1 = 'Q1'
                EN2 = 'Q2'
            }
            # [ 阶段步骤 ] 定义本阶段要执行的脚本
            steps {
            
                echo 'Hello World'
                echo "环境变量 EN1 = ${env.EN1} EN2 = ${env.EN2}"
                echo "用户参数 姓名: ${params.Name} 年龄 = ${params.Age}"
                
                sh 'echo "环境变量 EN1 = ${EN1} EN2 = ${EN2}"'
                
                script {
                    env.Name = params.Name
                    env.Age = params.Age
                }
                sh 'echo "用户参数 姓名: ${Name} 年龄 = ${Age}"'
                
            }
        }
        
        stage('第二阶段') {
            steps {
               ......
            }
        }
        
        stage('第三阶段：条件阶段') {
        
            when {
    			environment name: 'EN1', value: 'Q1'
			}
            steps {
               ......
            }
        }
        
    }
    
}


```
